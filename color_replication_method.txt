Methodology Automotive Paint Simulation via CSSStandard web colors use a single flat hex code, but real automotive paint (and Forza's rendering engine) relies on complex light interactions. To accurately display these on forza-color-repo.vercel.app, the team must use layered CSS backgrounds to simulate a 3D environment.Here is the breakdown of the rendering logicThe Base Coat (color1) This is the underlying diffuse color of the car. It is set as the foundational background-color.  The Specular Hit  Flake (color2) For Metal Flake, Pearlescent, and Two-Tone paints, the second color only appears where light directly hits the pigment. This is simulated using an off-center CSS radial-gradient acting as a localized light bounce.  The Clearcoat (Gloss) Glossy paints (Normal, Metal Flake) have a transparent, highly reflective top layer. This is simulated using a sharp, diagonal linear-gradient of semi-transparent white across the top edge.Panel Curvature (Shadow) To prevent the colors from looking flat, a dark radial-gradient or linear-gradient is pushed to the bottom of the swatch, mimicking how a car door curves away from the sun.Carbon Fiber Relies on a micro-texture. This is generated procedurally using crisscrossing repeating-linear-gradient functions to simulate the physical weave.Implementation CodeProvide this single-file HTMLJSCSS solution to Antigravity.          Core Math Convert Forza 0-1 HSB to Standard RGB
        function hsvToRgb(h, s, v) {
            let r, g, b;
            let i = Math.floor(h  6);
            let f = h  6 - i;
            let p = v  (1 - s);
            let q = v  (1 - f  s);
            let t = v  (1 - (1 - f)  s);
            switch (i % 6) {
                case 0 r = v, g = t, b = p; break;
                case 1 r = q, g = v, b = p; break;
                case 2 r = p, g = v, b = t; break;
                case 3 r = p, g = q, b = v; break;
                case 4 r = t, g = p, b = v; break;
                case 5 r = v, g = p, b = q; break;
            }
            return { r Math.round(r  255), g Math.round(g  255), b Math.round(b  255) };
        }

         Convert RGB to HEX
        function rgbToHex(r, g, b) {
            return # + (1  24  r  16  g  8  b).toString(16).slice(1).toUpperCase();
        }

         Advanced Material Shader Generator
        function getAdvancedMaterialCSS(item) {
            const hex1 = rgbToHex(hsvToRgb(item.color1.h, item.color1.s, item.color1.b).r, hsvToRgb(item.color1.h, item.color1.s, item.color1.b).g, hsvToRgb(item.color1.h, item.color1.s, item.color1.b).b);
            
            let hex2 = hex1;
            if (item.color2) {
                const rgb2 = hsvToRgb(item.color2.h, item.color2.s, item.color2.b);
                hex2 = rgbToHex(rgb2.r, rgb2.g, rgb2.b);
            }
            
            const type = (item.colorType  ).toLowerCase();

             1. MATTE & TWO-TONE MATTE
            if (type.includes('matte')) {
                return `
                    background-color ${hex1};
                    background-image 
                        radial-gradient(circle at 50% 30%, ${hex2} 0%, transparent 75%),
                        linear-gradient(180deg, rgba(255,255,255,0.05) 0%, rgba(0,0,0,0.3) 100%);
                `;
            } 
            
             2. METAL FLAKE & PEARLESCENT & TWO-TONE POLISHED
            else if (type.includes('flake')  type.includes('two-tone')  type.includes('pearl')) {
                return `
                    background-color ${hex1};
                    background-image 
                        radial-gradient(ellipse at 40% 35%, ${hex2} 0%, transparent 65%),
                        linear-gradient(105deg, rgba(255,255,255,0.5) 0%, rgba(255,255,255,0.1) 12%, transparent 20%),
                        radial-gradient(circle at 50% 150%, rgba(0,0,0,0.8) 0%, transparent 80%);
                `;
            } 
            
             3. CARBON FIBER
            else if (type.includes('carbon')) {
                const isGlossy = type.includes('polished');
                const highlight = isGlossy  'rgba(255,255,255,0.25)'  'rgba(255,255,255,0.05)';
                return `
                    background-color ${hex1};
                    background-image 
                        radial-gradient(circle at 40% 30%, ${highlight} 0%, transparent 70%),
                        repeating-linear-gradient(45deg, transparent, transparent 3px, rgba(0,0,0,0.4) 3px, rgba(0,0,0,0.4) 6px),
                        repeating-linear-gradient(-45deg, transparent, transparent 3px, rgba(0,0,0,0.4) 3px, rgba(0,0,0,0.4) 6px);
                `;
            } 
            
             4. NORMAL (Standard Gloss)
            else {
                return `
                    background-color ${hex1};
                    background-image 
                        linear-gradient(105deg, rgba(255,255,255,0.4) 0%, rgba(255,255,255,0.05) 15%, transparent 22%),
                        linear-gradient(180deg, transparent 60%, rgba(0,0,0,0.5) 100%);
                `;
            }
        }
